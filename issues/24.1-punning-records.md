```cpl
let c: int = 42;
let rec: (c: int) = (

);

	$c, % sugar for `c= c`

```
```cpl
($a, b= 42); % sugar for `(a= a, b= 42)`
[$a, b= 42]; % sugar for `[a= a, b= 42]`
```
